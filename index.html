CREATE INDEX IF NOT EXISTS idx_stores_place_id ON stores(place_id); -- ============================================ -- 2. store_keywords 테이블 (매장별 추적 키워드) -- ============================================ CREATE TABLE IF NOT EXISTS store_keywords ( id BIGSERIAL PRIMARY KEY, store_id BIGINT NOT NULL REFERENCES stores(id) ON DELETE CASCADE, keyword TEXT NOT NULL, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), UNIQUE(store_id, keyword) ); CREATE INDEX IF NOT EXISTS idx_store_keywords_store ON store_keywords(store_id); -- ============================================ -- 3. store_blog_posts 테이블 (플레이스 비공식 API 결과 캐시) -- 매장에 연결된 블로그 리뷰 URL 목록. 24h마다 갱신. -- ============================================ CREATE TABLE IF NOT EXISTS store_blog_posts ( id BIGSERIAL PRIMARY KEY, store_id BIGINT NOT NULL REFERENCES stores(id) ON DELETE CASCADE, blog_url TEXT NOT NULL, -- 정규화된 블로그 포스트 URL blog_id TEXT, -- 추출된 blog_id (매칭 가속용) post_id TEXT, -- 추출된 post_id (매칭 가속용) title TEXT, fetched_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), UNIQUE(store_id, blog_url) ); CREATE INDEX IF NOT EXISTS idx_sbp_store ON store_blog_posts(store_id); CREATE INDEX IF NOT EXISTS idx_sbp_blog_post ON store_blog_posts(blog_id, post_id); CREATE INDEX IF NOT EXISTS idx_sbp_fetched_at ON store_blog_posts(fetched_at); -- ============================================ -- 4. store_place_fetches 테이블 (마지막 플레이스 API 호출 기록 - 캐시 만료 판단용) -- ============================================ CREATE TABLE IF NOT EXISTS store_place_fetches ( store_id BIGINT PRIMARY KEY REFERENCES stores(id) ON DELETE CASCADE, fetched_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), post_count INTEGER NOT NULL DEFAULT 0, success BOOLEAN NOT NULL DEFAULT TRUE, error_msg TEXT ); -- ============================================ -- 5. store_rankings 테이블 (매장별 키워드 일일 순위 기록) -- 기존 rank_history는 포스팅 순위 조회 페이지에서 계속 사용, 건드리지 않음 -- ============================================ CREATE TABLE IF NOT EXISTS store_rankings ( id BIGSERIAL PRIMARY KEY, store_id BIGINT NOT NULL REFERENCES stores(id) ON DELETE CASCADE, keyword TEXT NOT NULL, checked_date DATE NOT NULL DEFAULT CURRENT_DATE, rank INTEGER, -- 1~10위 또는 NULL (10위 밖) matched_blog_url TEXT, -- 실제 매칭된 포스트 URL matched_title TEXT, -- 해당 포스트 제목 search_volume INTEGER, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), UNIQUE(store_id, keyword, checked_date) -- 같은 날 같은 키워드는 1건만 (덮어쓰기) ); CREATE INDEX IF NOT EXISTS idx_store_rankings_lookup ON store_rankings(store_id, keyword, checked_date DESC); -- ============================================ -- 확인용 쿼리 (실행 후 테이블이 제대로 생성됐는지 확인) -- ============================================ -- SELECT table_name FROM information_schema.tables -- WHERE table_schema = 'public' -- ORDER BY table_name;
